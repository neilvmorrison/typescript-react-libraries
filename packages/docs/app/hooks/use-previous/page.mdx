# usePrevious

Track the previous value of a prop or state.

## Usage

```typescript
import { usePrevious } from '@libraries/hooks';

export function ValueDisplay({ value }: { value: number }) {
  const previousValue = usePrevious(value);

  return (
    <div>
      <p>Current: {value}</p>
      <p>Previous: {previousValue}</p>
    </div>
  );
}
```

## API Reference

### Import

```typescript
import { usePrevious } from '@libraries/hooks';
```

### Parameters

| Name    | Type | Required | Description        | Default |
| ------- | ---- | -------- | ------------------ | ------- |
| `value` | `T`  | Yes      | The value to track | N/A     |

### Returns

| Name          | Type             | Description                                                         |
| ------------- | ---------------- | ------------------------------------------------------------------- |
| previousValue | `T \| undefined` | The previous value from the last render (undefined on first render) |
