# TextInput

Simple text input component that extends standard HTML input elements. Supports all HTML input attributes and types.

## Usage

```typescript
import { TextInput } from '@libraries/component-library';

export function Example() {
  const [value, setValue] = useState('');

  return (
    <div>
      <TextInput
        type="email"
        placeholder="Enter your email"
        value={value}
        onChange={(e) => setValue(e.target.value)}
      />
      <TextInput
        type="password"
        placeholder="Enter your password"
      />
      <TextInput
        placeholder="Enter your username"
      />
    </div>
  );
}
```

## API Reference

### Import

```typescript
import { TextInput } from '@libraries/component-library';
```

### Props

| Name       | Type                                    | Required | Description                       | Default  |
| ---------- | --------------------------------------- | -------- | --------------------------------- | -------- |
| `type`     | `string`                                | No       | HTML input type                   | `'text'` |
| `ref`      | `React.Ref<HTMLInputElement>`           | No       | Forward ref to HTML input element | N/A      |
| `...props` | `InputHTMLAttributes<HTMLInputElement>` | No       | Standard HTML input attributes    | N/A      |


### With Ref

```typescript
import { useRef } from 'react';

export function TextInputWithRef() {
  const inputRef = useRef<HTMLInputElement>(null);

  return (
    <TextInput
      ref={inputRef}
      placeholder="Focus me"
      onFocus={() => inputRef.current?.select()}
    />
  );
}
```
